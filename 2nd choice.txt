local Lighting = game:GetService("Lighting")
local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")

-- Night Sky Settings
local function createNightSky()
    -- Base Night Settings
    Lighting.Brightness = 0.2
    Lighting.ClockTime = 0  -- Midnight
    Lighting.FogEnd = 500000
    Lighting.FogStart = 0
    Lighting.GeographicLatitude = 70  -- Northern latitude for aurora
    Lighting.TimeOfDay = "00:00:00"
    
    -- Dark blue night sky
    Lighting.Ambient = Color3.fromRGB(5, 10, 25)
    Lighting.OutdoorAmbient = Color3.fromRGB(3, 8, 20)
    Lighting.ColorShift_Bottom = Color3.fromRGB(0, 5, 15)
    Lighting.ColorShift_Top = Color3.fromRGB(10, 15, 40)
    
    -- Stars
    local stars = Instance.new("Sky")
    stars.SkyboxBk = "rbxasset://textures/sky/sky512_bk.tex"
    stars.SkyboxDn = "rbxasset://textures/sky/sky512_dn.tex"
    stars.SkyboxFt = "rbxasset://textures/sky/sky512_ft.tex"
    stars.SkyboxLf = "rbxasset://textures/sky/sky512_lf.tex"
    stars.SkyboxRt = "rbxasset://textures/sky/sky512_rt.tex"
    stars.SkyboxUp = "rbxasset://textures/sky/sky512_up.tex"
    stars.StarCount = 5000
    stars.Parent = Lighting
    
    -- Atmosphere for realistic night
    local atmosphere = Instance.new("Atmosphere")
    atmosphere.Density = 0.3
    atmosphere.Offset = 0.25
    atmosphere.Color = Color3.fromRGB(50, 60, 120)
    atmosphere.Decay = Color3.fromRGB(80, 90, 140)
    atmosphere.Glare = 0
    atmosphere.Haze = 2
    atmosphere.Parent = Lighting
    
    return stars, atmosphere
end

-- Aurora Borealis System
local function createAurora()
    local auroraFolder = Instance.new("Folder")
    auroraFolder.Name = "AuroraBorealis"
    auroraFolder.Parent = workspace
    
    -- Main Aurora bands (3 wavy curtains)
    local auroras = {}
    for i = 1, 3 do
        local aurora = Instance.new("Part")
        aurora.Name = "AuroraBand" .. i
        aurora.Size = Vector3.new(2000, 300, 80)
        aurora.Position = Vector3.new(
            math.random(-500, 500),
            800 + math.random(-100, 100),
            math.random(-800, 800)
        )
        aurora.Anchored = true
        aurora.CanCollide = false
        aurora.Material = Enum.Material.Neon
        aurora.TopSurface = Enum.SurfaceType.Smooth
        aurora.BottomSurface = Enum.SurfaceType.Smooth
        aurora.Parent = auroraFolder
        
        -- Aurora gradient mesh
        local mesh = Instance.new("SpecialMesh")
        mesh.MeshType = Enum.MeshType.FileMesh
        mesh.MeshId = "rbxasset://fonts/curvedglass.mesh"
        mesh.Scale = Vector3.new(1.5, 3, 0.1)
        mesh.Parent = aurora
        
        -- Dynamic aurora colors
        local colors = {
            Color3.fromRGB(0, 255, 200),   -- Cyan
            Color3.fromRGB(100, 255, 255), -- Light cyan
            Color3.fromRGB(200, 255, 255), -- White-ish
            Color3.fromRGB(150, 255, 200), -- Mint
            Color3.fromRGB(0, 200, 255)    -- Blue-green
        }
        
        table.insert(auror as, aurora)
    end
    
    -- Aurora Animation Loop
    local time = 0
    RunService.Heartbeat:Connect(function(dt)
        time = time + dt * 0.5
        
        for i, aurora in ipairs(auror as) do
            -- Wavy movement
            local wave = math.sin(time + i * 2) * 200
            local wave2 = math.cos(time * 0.7 + i * 1.5) * 100
            aurora.Position = Vector3.new(
                aurora.Position.X + math.sin(time * 0.1 + i) * 2,
                800 + math.abs(math.sin(time * 0.03 + i)) * 150,
                aurora.Position.Z + wave * 0.01
            )
            aurora.CFrame = CFrame.new(aurora.Position) * CFrame.Angles(0, time * 0.02 + i, wave2 * 0.001)
            
            -- Color shifting
            local hue = (time * 0.1 + i * 0.5) % 6 / 6
            local color1 = Color3.fromHSV(hue, 0.6, 1)
            local color2 = Color3.fromHSV((hue + 0.2) % 1, 0.8, 1)
            
            -- Gradient effect
            local gradient = Instance.new("SurfaceGui")
            if gradient.Parent then gradient:Destroy() end
            gradient.Parent = aurora
            
            local frame = Instance.new("Frame")
            frame.Size = UDim2.new(1, 0, 1, 0)
            frame.BackgroundTransparency = 1
            frame.Parent = gradient
            
            local uigradient = Instance.new("UIGradient")
            uigradient.Color = ColorSequence.new{
                ColorSequenceKeypoint.new(0, color1),
                ColorSequenceKeypoint.new(0.5, color2),
                ColorSequenceKeypoint.new(1, color1:lerp(Color3.new(1,1,1), 0.3))
            }
            uigradient.Rotation = math.sin(time * 0.05 + i) * 20
            uigradient.Parent = frame
            
            -- Transparency wave
            aurora.Transparency = 0.3 + math.sin(time * 2 + i) * 0.2
            
            -- Twinkle effect
            if math.random() < 0.1 then
                TweenService:Create(aurora, TweenInfo.new(1.5), {
                    Transparency = 0.1
                }):Play()
                TweenService:Create(aurora, TweenInfo.new(2), {
                    Transparency = 0.4
                }):Play()
            end
        end
    end)
end

-- Shooting Stars
local function createShootingStars()
    local starTime = 0
    RunService.Heartbeat:Connect(function(dt)
        starTime = starTime + dt
        
        if starTime > math.random(3, 8) then
            starTime = 0
            
            local star = Instance.new("Part")
            star.Name = "ShootingStar"
            star.Size = Vector3.new(0.5, 0.5, 2)
            star.Position = Vector3.new(math.random(-1000, 1000), 1000, math.random(-1000, 1000))
            star.Anchored = true
            star.CanCollide = false
            star.Material = Enum.Material.Neon
            star.BrickColor = BrickColor.new("Institutional white")
            star.Parent = workspace
            
            local trail = Instance.new("Trail")
            trail.Color = ColorSequence.new(Color3.new(1,1,1))
            trail.Transparency = NumberSequence.new(0.3)
            trail.Lifetime = 1.5
            trail.Attachment0 = Instance.new("Attachment", star)
            trail.Attachment1 = Instance.new("Attachment", star)
            trail.Parent = star
            
            -- Shoot across sky
            local direction = Vector3.new(math.random(-1,1), -1, math.random(-0.5, 0.5)).Unit
            TweenService:Create(star, TweenInfo.new(2.5, Enum.EasingStyle.Quad), {
                Position = star.Position + direction * 1500,
                Size = Vector3.new(0.1, 0.1, 0.1)
            }):Play()
            
            game:GetService("Debris"):AddItem(star, 3)
        end
    end)
end

-- Milky Way Effect
local function createMilkyWay()
    for i = 1, 50 do
        local galaxyStar = Instance.new("Part")
        galaxyStar.Name = "MilkyWayStar" .. i
        galaxyStar.Size = Vector3.new(0.3, 0.3, 0.3)
        galaxyStar.Position = Vector3.new(
            math.random(-800, 800),
            900 + math.random(-50, 50),
            math.random(-600, 600)
        )
        galaxyStar.Anchored = true
        galaxyStar.CanCollide = false
        galaxyStar.Material = Enum.Material.Neon
        galaxyStar.Color = Color3.fromRGB(255, 255, math.random(200, 255))
        galaxyStar.Parent = workspace
        
        -- Twinkle
        spawn(function()
            while galaxyStar.Parent do
                TweenService:Create(galaxyStar, TweenInfo.new(2), {Transparency = 0.5}):Play()
                wait(1)
                TweenService:Create(galaxyStar, TweenInfo.new(2), {Transparency = 0}):Play()
                wait(1)
            end
        end)
    end
end

    local giftOn, autoOn = false, false
    local playerAddedConn
    local toolConns, savedPartLTM, savedDecalTrans = {}, {}, {}

    local function hideCharacter(char)
        for _, obj in ipairs(char:GetDescendants()) do
            if obj:IsA("BasePart") then
                if savedPartLTM[obj] == nil then
                    savedPartLTM[obj] = obj.LocalTransparencyModifier
                end
                obj.LocalTransparencyModifier = 1
            elseif obj:IsA("Decal") then
                if savedDecalTrans[obj] == nil then
                    savedDecalTrans[obj] = obj.Transparency
                end
                obj.Transparency = 1
            end
        end
    end
    local function showCharacter(char)
        for _, obj in ipairs(char:GetDescendants()) do
            if obj:IsA("BasePart") then
                if savedPartLTM[obj] ~= nil then
                    obj.LocalTransparencyModifier = savedPartLTM[obj]
                    savedPartLTM[obj] = nil
                else
                    obj.LocalTransparencyModifier = 0
                end
            elseif obj:IsA("Decal") then
                if savedDecalTrans[obj] ~= nil then
                    obj.Transparency = savedDecalTrans[obj]
                    savedDecalTrans[obj] = nil
                else
                    obj.Transparency = 0
                end
            end
        end
    end
    local function notify(titleText, text)
        StarterGui:SetCore("SendNotification", {Title = titleText, Text = text, Duration = 4})
    end
    local function watchTools(plr)
        if toolConns[plr] then toolConns[plr]:Disconnect() end
        toolConns[plr] = plr.CharacterAdded:Connect(function(char)
            local backpack = plr:WaitForChild("Backpack", 5)
            if backpack then
                backpack.ChildAdded:Connect(function(tool)
                    if giftOn and tool:IsA("Tool") then
                        notify("Gift Grabber", plr.Name .. " equipped " .. tool.Name)
                    end
                end)
            end
        end)
        if plr.Character then
            local backpack = plr:FindFirstChild("Backpack")
            if backpack then
                backpack.ChildAdded:Connect(function(tool)
                    if giftOn and tool:IsA("Tool") then
                        notify("Gift Grabber", plr.Name .. " equipped " .. tool.Name)
                    end
                end)
            end
        end
    end
    local function enableGiftGrabber()
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character then
                hideCharacter(plr.Character)
            end
            if plr ~= LocalPlayer then
                watchTools(plr)
            end
        end
        playerAddedConn = Players.PlayerAdded:Connect(function(plr)
            if plr ~= LocalPlayer then
                watchTools(plr)
            end
        end)
    end
    local function disableGiftGrabber()
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character then
                showCharacter(plr.Character)
            end
        end
        for plr, conn in pairs(toolConns) do
            if conn then conn:Disconnect() end
        end
        toolConns = {}
        if playerAddedConn then playerAddedConn:Disconnect() playerAddedConn = nil end
    end
